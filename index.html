Click to Play through steps
	var $sequence = $('.sequence'),
	mySequence = new SuperSequence($sequence, 40, 30, "frame"),
	nextTar = 0,
	checkpoints = [10, 18, 26, 40];

	$sequence.click(function(){
		//playTo a frame in the sequence
		mySequence.playTo(checkpoints[nextTar]);

		//increment our counter. Next click we'll go to 18 (and so on)
		nextTar++;
		if(nextTar > checkpoints.length){
			mySequence.playTo(1); //we can play backward too
			nextTar = 0;
		}
	});

	//Want to know when a frame has been reached? Do it like this
	mySequence.on(15, function(e){
		console.log(e.label); //frame15
	});
